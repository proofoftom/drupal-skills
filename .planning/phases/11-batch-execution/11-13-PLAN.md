**Part A: Validate all 13 benchmark.json files**

```bash
PHASE_DIR=".planning/phases/11-batch-execution"
SKILLS=(
  drupal-access-security
  drupal-batch-queue-cron
  drupal-caching
  drupal-config-storage
  drupal-database-api
  drupal-entities-fields
  drupal-forms-api
  drupal-module-scaffold
  drupal-plugins-blocks
  drupal-routing-controllers
  drupal-testing
  drupal-theming
  drupal-views-dev
)

VALID=0
INVALID=0
for skill in "${SKILLS[@]}"; do
  bm="$PHASE_DIR/workspaces/${skill}-workspace/iteration-1/benchmark.json"
  if [ -f "$bm" ] && jq -e 'has("metadata") and has("runs") and has("run_summary") and (.run_summary | has("delta"))' "$bm" > /dev/null 2>&1; then
    VALID=$((VALID + 1))
  else
    INVALID=$((INVALID + 1))
    echo "INVALID: $skill"
  fi
done
echo "Valid: $VALID / Invalid: $INVALID / Total: $((VALID + INVALID))"
```

All 13 must be valid. If any are invalid, investigate and fix before proceeding.

**Part B: Extract results and build consolidated table**

For each skill, extract with_skill pass_rate, without_skill pass_rate, delta, and source.

```bash
for skill in "${SKILLS[@]}"; do
  bm="$PHASE_DIR/workspaces/${skill}-workspace/iteration-1/benchmark.json"
  WITH=$(jq '.run_summary.with_skill.pass_rate.mean' "$bm")
  WITHOUT=$(jq '.run_summary.without_skill.pass_rate.mean' "$bm")
  DELTA=$(jq '.run_summary.delta.pass_rate' "$bm")
  TOTAL=$(jq '.runs[0].result.total' "$bm")
  echo "$skill | $WITH | $WITHOUT | $DELTA | $TOTAL expectations"
done
```
