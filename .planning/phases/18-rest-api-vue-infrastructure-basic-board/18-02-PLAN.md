**Step 4: Create Drupal.behaviors bridge** (`modules/group_ai_pm/js/src/main.js`):

This is the entry point. It wraps the Vue app in Drupal.behaviors with once() guard:

```javascript
(function (Drupal, once, drupalSettings) {
  'use strict';

  Drupal.behaviors.groupAiPmKanban = {
    attach: function (context) {
      once('kanban-app', '#kanban-app', context).forEach(function (element) {
        var config = drupalSettings.groupAiPm.kanban;
        var app = Vue.createApp(GroupAiPmKanban.KanbanBoard, {
          projectId: config.projectId,
          columns: config.columns,
          initialTasks: config.tasks,
          apiBaseUrl: config.apiBaseUrl,
          csrfTokenUrl: config.csrfTokenUrl,
          permissions: config.permissions,
          statusLabels: config.statusLabels,
          priorityLabels: config.priorityLabels,
        });
        app.mount(element);
        element._vueApp = app;
      });
    },
    detach: function (context, settings, trigger) {
      if (trigger === 'unload') {
        var elements = (context.querySelectorAll)
          ? context.querySelectorAll('[data-once="kanban-app"]')
          : [];
        elements.forEach(function (element) {
          if (element._vueApp) {
            element._vueApp.unmount();
            element._vueApp = null;
          }
        });
      }
    },
  };
})(Drupal, once, drupalSettings);
```

WAIT -- this pattern does NOT work with Vite library mode. The IIFE wrapper from Vite will encapsulate the module. The behaviors bridge needs to be OUTSIDE the Vue component tree but INSIDE the IIFE output.

Correct approach: main.js is the Vite entry point. It imports the root Vue component and exports it. The IIFE exposes it as `GroupAiPmKanban`. Then a SEPARATE non-bundled JS file (or inline in main.js before the Vue import) handles Drupal.behaviors.

Actually, the cleaner approach per the research: main.js IS the entry point that does everything. It imports KanbanBoard.vue, and the Drupal.behaviors code lives in main.js. Vite compiles it all into kanban.js IIFE. The IIFE accesses `Drupal`, `once`, `drupalSettings` from globals (they are NOT imported, so Vite does not need to externalize them). Only `vue` is externalized because Vue SFCs use `import { ref } from 'vue'`.

Revised main.js approach:
```javascript
import KanbanBoard from './components/KanbanBoard.vue';

(function (Drupal, once, drupalSettings) {
  'use strict';

  Drupal.behaviors.groupAiPmKanban = {
    attach: function (context) {
      once('kanban-app', '#kanban-app', context).forEach(function (element) {
        var config = drupalSettings.groupAiPm.kanban;
        var app = Vue.createApp(KanbanBoard, {
          projectId: config.projectId,
          columns: config.columns,
          initialTasks: config.tasks,
          apiBaseUrl: config.apiBaseUrl,
          csrfTokenUrl: config.csrfTokenUrl,
          permissions: config.permissions,
          statusLabels: config.statusLabels,
          priorityLabels: config.priorityLabels,
        });
        app.mount(element);
        element._vueApp = app;
      });
    },
    detach: function (context, settings, trigger) {
      if (trigger === 'unload') {
        var elements = (context.querySelectorAll)
          ? context.querySelectorAll('[data-once="kanban-app"]')
          : [];
        elements.forEach(function (element) {
          if (element._vueApp) {
            element._vueApp.unmount();
            element._vueApp = null;
          }
        });
      }
    },
  };
})(Drupal, once, drupalSettings);
```

`Drupal`, `once`, `drupalSettings` are global variables provided by Drupal core JS. They do NOT need Vite externalization because they are accessed as function parameters from the global scope, not via `import` statements. `Vue` is also a global (from vue.global.prod.js) but SFC `import { ref } from 'vue'` requires Vite externalization.

**Step 5: Create CSRF-aware API wrapper** (`modules/group_ai_pm/js/src/api/drupal.js`):

```javascript
let csrfToken = null;

async function ensureCsrfToken(csrfTokenUrl) {
  if (!csrfToken) {
    const response = await fetch(csrfTokenUrl);
    csrfToken = await response.text();
  }
  return csrfToken;
}

export async function apiGet(url) {
  const response = await fetch(url + '?_format=json');
  if (!response.ok) {
    const error = new Error('API error: ' + response.status);
    error.status = response.status;
    throw error;
  }
  return response.json();
}

export async function apiMutate(method, url, data, csrfTokenUrl) {
  const token = await ensureCsrfToken(csrfTokenUrl);
  const response = await fetch(url + '?_format=json', {
    method: method,
    headers: {
      'Content-Type': 'application/json',
      'X-CSRF-Token': token,
    },
    body: JSON.stringify(data),
  });
  if (response.status === 403) {
    csrfToken = null;
    const retryToken = await ensureCsrfToken(csrfTokenUrl);
    const retry = await fetch(url + '?_format=json', {
      method: method,
      headers: {
        'Content-Type': 'application/json',
        'X-CSRF-Token': retryToken,
      },
      body: JSON.stringify(data),
    });
    if (!retry.ok) {
      const error = new Error('API error: ' + retry.status);
      error.status = retry.status;
      throw error;
    }
    return retry.json();
  }
  if (!response.ok) {
    const error = new Error('API error: ' + response.status);
    error.status = response.status;
    throw error;
  }
  return response.json();
}

export function clearCsrfToken() {
  csrfToken = null;
}
```
