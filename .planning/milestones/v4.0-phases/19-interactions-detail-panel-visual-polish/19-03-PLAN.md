    **Filtered columns computed:**
    ```javascript
    const filteredColumns = computed(() => {
      const result = {};
      for (const [status, tasks] of Object.entries(columns.value)) {
        result[status] = tasks.filter(task => {
          if (filters.value.assignee && (!task.assignee || String(task.assignee.id) !== String(filters.value.assignee))) {
            return false;
          }
          if (filters.value.priority && task.priority !== filters.value.priority) {
            return false;
          }
          return true;
        });
      }
      return result;
    });
    ```
    Pass `filteredColumns` to KanbanColumn instead of raw `columns`.
